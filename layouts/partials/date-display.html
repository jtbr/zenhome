{{- /* Renders a element. Supports partial dates via the `partial_date` front matter field: if set to a string in "YYYY" or "YYYY-MM" format, only the given precision is shown. (Kept separate from the reserved `date` field because Hugo >= 0.135.0 hard-fails the build if `date` isn't a fully parsable date.) Context: dict with "item" (page) and optional "long" (bool, default false for short month name). */ -}} {{- $item := .item -}} {{- $long := .long | default false -}} {{- $rawDate := "" -}} {{- with $item.Params.partial_date -}}{{- $rawDate = printf "%v" . -}}{{- end -}} {{- if findRE `^\d{4}$` $rawDate -}} {{ $rawDate }} {{- else if findRE `^\d{4}-\d{2}$` $rawDate -}} {{- $t := time (printf "%s-01" $rawDate) -}} {{ if $long }}{{ $t.Format "January 2006" }}{{ else }}{{ $t.Format "Jan 2006" }}{{ end }} {{- else if not $item.Date.IsZero -}} {{- if $long -}} {{ $item.Date.Format "January 2, 2006" }} {{- else -}} {{ $item.Date.Format "Jan 2, 2006" }} {{- end -}} {{- end -}}
